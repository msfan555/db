UPDATE 更新

  • UPDATE `students` SET `uni_id`='915084', `seat_num`='11042' WHERE `id`='480';

  • UPDATE `students` SET `uni_id`='915084',`seat_num`='11042'; //沒有加WHERE條件造成整個資料表欄位都更新成一樣內容

  • UPDATE `students` SET `major`='商業貿易科' WHERE `major`='國際貿易科'; //符合條件的資料表欄位更新成相同結果

DELETE 刪除

  • DELETE FROM `students` WHERE `id`='10';

  • DELETE FROM `students` WHERE `major`='商業貿易科'; //全部符合條件的資料都刪除

  • DELETE FROM `students` //沒加WHERE條件，全部資料都刪除

SELECT 查詢

  • SELECT * FROM `students` WHERE `id` < 20;

  • SELECT * FROM `students` WHERE `id`<100 OR `id`>200;

  • SELECT * FROM `students` WHERE `id`>100 AND `id`< 200;

  • SELECT `id` as '序號',`name` as '姓名' FROM `students` WHERE `id`<100 OR `id`>200;
